### 2.1 Vector/Embedding 类型

#### 2.1.1 类型定义

```rust
/// 向量类型 - 用于 AI/ML 应用
///
/// 支持存储机器学习模型生成的向量嵌入，用于语义搜索、推荐系统等场景
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Encode, Decode)]
pub struct Vector {
    /// 向量维度
    pub dimensions: u32,
    /// 向量值（通常为 f32 或 f64）
    pub values: Vec<f32>,
    /// 生成向量的模型名称（可选）
    pub model: Option<String>,
    /// 向量数据类型标识
    pub data_type: VectorDataType,
}

/// 向量数据类型
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash, Encode, Decode)]
pub enum VectorDataType {
    Float32,
    Float64,
    Int8,
    Int16,
    Int32,
}

impl Vector {
    /// 创建新向量
    pub fn new(values: Vec<f32>) -> Self {
        let dimensions = values.len() as u32;
        Self {
            dimensions,
            values,
            model: None,
            data_type: VectorDataType::Float32,
        }
    }

    /// 计算余弦相似度
    pub fn cosine_similarity(&self, other: &Vector) -> Result<f64, String> {
        if self.dimensions != other.dimensions {
            return Err("向量维度不匹配".to_string());
        }

        let dot_product: f32 = self.values.iter()
            .zip(other.values.iter())
            .map(|(a, b)| a * b)
            .sum();

        let norm_a: f32 = self.values.iter().map(|x| x * x).sum::<f32>().sqrt();
        let norm_b: f32 = other.values.iter().map(|x| x * x).sum::<f32>().sqrt();

        if norm_a == 0.0 || norm_b == 0.0 {
            return Ok(0.0);
        }

        Ok((dot_product / (norm_a * norm_b)) as f64)
    }

    /// 计算欧几里得距离
    pub fn euclidean_distance(&self, other: &Vector) -> Result<f64, String> {
        if self.dimensions != other.dimensions {
            return Err("向量维度不匹配".to_string());
        }

        let sum_sq: f64 = self.values.iter()
            .zip(other.values.iter())
            .map(|(a, b)| (a - b).powi(2) as f64)
            .sum();

        Ok(sum_sq.sqrt())
    }

    /// 估算内存使用大小
    pub fn estimated_size(&self) -> usize {
        std::mem::size_of::<Self>()
            + self.values.capacity() * std::mem::size_of::<f32>()
            + self.model.as_ref().map(|s| s.capacity()).unwrap_or(0)
    }
}

impl Default for Vector {
    fn default() -> Self {
        Self {
            dimensions: 0,
            values: Vec::new(),
            model: None,
            data_type: VectorDataType::Float32,
        }
    }
}

impl std::fmt::Display for Vector {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "Vector(dim={}, values=[", self.dimensions)?;
        for (i, val) in self.values.iter().enumerate() {
            if i > 0 {
                write!(f, ", ")?;
            }
            write!(f, "{:.4}", val)?;
        }
        write!(f, "])")
    }
}
```

### 3.1 Regex 类型

#### 3.1.1 类型定义

```rust
/// 正则表达式类型
///
/// 用于模式匹配查询和数据验证
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash, Encode, Decode)]
pub struct Regex {
    /// 正则表达式模式
    pub pattern: String,
    /// 正则表达式选项（如 i, g, m 等）
    pub options: String,
}

impl Regex {
    /// 创建新的正则表达式
    pub fn new(pattern: &str, options: &str) -> Self {
        Self {
            pattern: pattern.to_string(),
            options: options.to_string(),
        }
    }

    /// 编译为 Rust Regex
    pub fn compile(&self) -> Result<regex::Regex, String> {
        let pattern = self.build_pattern();
        regex::Regex::new(&pattern)
            .map_err(|e| format!("正则表达式编译失败: {}", e))
    }

    /// 构建完整的正则表达式字符串
    fn build_pattern(&self) -> String {
        let mut pattern = String::new();

        if self.options.contains('i') {
            pattern.push_str("(?i)");
        }

        pattern.push_str(&self.pattern);

        pattern
    }

    /// 测试字符串是否匹配
    pub fn is_match(&self, text: &str) -> Result<bool, String> {
        let regex = self.compile()?;
        Ok(regex.is_match(text))
    }

    /// 查找所有匹配
    pub fn find_all(&self, text: &str) -> Result<Vec<String>, String> {
        let regex = self.compile()?;
        Ok(regex.find_iter(text)
            .map(|m| m.as_str().to_string())
            .collect())
    }

    /// 估算内存使用大小
    pub fn estimated_size(&self) -> usize {
        std::mem::size_of::<Self>()
            + self.pattern.capacity()
            + self.options.capacity()
    }
}

impl Default for Regex {
    fn default() -> Self {
        Self {
            pattern: String::new(),
            options: String::new(),
        }
    }
}

impl std::fmt::Display for Regex {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "/{}/{}", self.pattern, self.options)
    }
}
```

### 3.2 ZonedDateTime 类型

#### 3.2.1 类型定义

```rust
/// 带时区的日期时间类型
///
/// 用于全球化应用，支持时区转换
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Encode, Decode)]
pub struct ZonedDateTime {
    pub year: i32,
    pub month: u32,
    pub day: u32,
    pub hour: u32,
    pub minute: u32,
    pub sec: u32,
    pub microsec: u32,
    /// 时区偏移（秒）
    pub timezone_offset: i32,
}

impl ZonedDateTime {
    /// 创建新的带时区的日期时间
    pub fn new(
        year: i32,
        month: u32,
        day: u32,
        hour: u32,
        minute: u32,
        sec: u32,
        microsec: u32,
        timezone_offset: i32,
    ) -> Self {
        Self {
            year,
            month,
            day,
            hour,
            minute,
            sec,
            microsec,
            timezone_offset,
        }
    }

    /// 转换为 UTC 时间
    pub fn to_utc(&self) -> DateTimeValue {
        let mut dt = DateTimeValue {
            year: self.year,
            month: self.month,
            day: self.day,
            hour: self.hour,
            minute: self.minute,
            sec: self.sec,
            microsec: self.microsec,
        };

        // 应用时区偏移
        dt.add_duration(&DurationValue {
            seconds: -self.timezone_offset as i64,
            microseconds: 0,
            months: 0,
        });

        dt
    }

    /// 从 UTC 时间创建
    pub fn from_utc(dt: DateTimeValue, timezone_offset: i32) -> Self {
        let mut zdt = Self {
            year: dt.year,
            month: dt.month,
            day: dt.day,
            hour: dt.hour,
            minute: dt.minute,
            sec: dt.sec,
            microsec: dt.microsec,
            timezone_offset,
        };

        // 应用时区偏移
        zdt.add_duration(&DurationValue {
            seconds: timezone_offset as i64,
            microseconds: 0,
            months: 0,
        });

        zdt
    }

    /// 添加持续时间
    pub fn add_duration(&mut self, duration: &DurationValue) {
        let mut dt = self.to_utc();
        dt.add_duration(duration);
        *self = Self::from_utc(dt, self.timezone_offset);
    }

    /// 减去持续时间
    pub fn sub_duration(&mut self, duration: &DurationValue) {
        let neg_duration = DurationValue {
            seconds: -duration.seconds,
            microseconds: -duration.microseconds,
            months: -duration.months,
        };
        self.add_duration(&neg_duration);
    }

    /// 估算内存使用大小
    pub fn estimated_size(&self) -> usize {
        std::mem::size_of::<Self>()
    }
}

impl Default for ZonedDateTime {
    fn default() -> Self {
        Self {
            year: 1970,
            month: 1,
            day: 1,
            hour: 0,
            minute: 0,
            sec: 0,
            microsec: 0,
            timezone_offset: 0,
        }
    }
}

impl std::fmt::Display for ZonedDateTime {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        let offset_hours = self.timezone_offset / 3600;
        let offset_minutes = (self.timezone_offset % 3600) / 60;
        let offset_sign = if self.timezone_offset >= 0 { '+' } else { '-' };
        write!(
            f,
            "{:04}-{:02}-{:02} {:02}:{:02}:{:02}.{:06} UTC{}{:02}:{:02}",
            self.year, self.month, self.day, self.hour, self.minute, self.sec, self.microsec,
            offset_sign, offset_hours.abs(), offset_minutes.abs()
        )
    }
}
```

#### 3.3.2 Polygon 类型

```rust
/// 多边形类型
///
/// 由一系列点组成的多边形（首尾相连）
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Encode, Decode)]
pub struct Polygon {
    pub points: Vec<GeographyValue>,
}

impl Polygon {
    pub fn new(points: Vec<GeographyValue>) -> Self {
        Self { points }
    }

    /// 计算多边形面积（使用鞋带公式）
    pub fn area(&self) -> f64 {
        if self.points.len() < 3 {
            return 0.0;
        }

        let mut area = 0.0;
        let n = self.points.len();

        for i in 0..n {
            let j = (i + 1) % n;
            let xi = self.points[i].longitude;
            let yi = self.points[i].latitude;
            let xj = self.points[j].longitude;
            let yj = self.points[j].latitude;
            area += (xj - xi) * (yi + yj);
        }

        area.abs() / 2.0
    }

    /// 检查点是否在多边形内（射线法）
    pub fn contains(&self, point: &GeographyValue) -> bool {
        let n = self.points.len();
        let mut inside = false;

        for i in 0..n {
            let j = (i + 1) % n;
            let xi = self.points[i].longitude;
            let yi = self.points[i].latitude;
            let xj = self.points[j].longitude;
            let yj = self.points[j].latitude;

            let intersect = ((yi > point.latitude) != (yj > point.latitude))
                && (point.longitude < (xj - xi) * (point.latitude - yi) / (yj - yi) + xi);

            if intersect {
                inside = !inside;
            }
        }

        inside
    }

    /// 估算内存使用大小
    pub fn estimated_size(&self) -> usize {
        std::mem::size_of::<Self>()
            + self.points.capacity() * std::mem::size_of::<GeographyValue>()
    }
}
```
